package com.interview;

import java.util.*;
import java.util.stream.Collectors;

public class Main {
    public static void main(String[] args) {
        mapaDeFrequencia();
    }

    private static void mapaDeFrequencia() {
        List listWords = Arrays.asList("java", "api", "exemplo", "stream", "java", "exemplo");
        Map mapWordsResp = listWords.stream()
            .collect(Collectors.groupingBy(w -> w, Collectors.counting()));
        System.out.println(mapWordsResp);
    }
}
